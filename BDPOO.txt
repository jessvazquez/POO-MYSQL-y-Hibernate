CREATE TABLE `esproyecto`.`empleados` ( `nombre` VARCHAR(100) NOT NULL , `edad` INT NOT NULL , `turno` VARCHAR(20) NOT NULL , `numeroEmpleado` INT NOT NULL , `area` VARCHAR(20) NOT NULL , `horaEntrada` VARCHAR(20) NOT NULL , `horaSalida` VARCHAR(20) NOT NULL , PRIMARY KEY (`numeroEmpleado`)) ENGINE = InnoDB;
